31       public void downloadFile (final Activity activity) { 
32           mgr = (DownloadManager) activity.getSystemService(DOWNLOAD_SERVICE); 
33           String file_url = serverURL + "files/test.xlsx"; 
34           Uri uri = Uri.parse(file_url); 
35           Toast.makeText(activity, "Downloading.", LENGTH_SHORT).show(); 
36           mgr.enqueue(new DownloadManager.Request(uri).setAllowedNetworkTypes(DownloadManager.Request.NETWORK_WIFI | DownloadManager.Request.NETWORK_MOBILE).setAllowedOverRoaming(false).setTitle("PrinLab") 
37                   .setDescription("Excel-file") 
38                   .setDestinationInExternalPublicDir(Environment.DIRECTORY_DOWNLOADS, "test.xlsx")); 
39       }

41       public void uploadFile(final Activity activity) { 
42           upLoadServerUri = serverURL + "upload.php"; 
43           HttpURLConnection conn; 
44           DataOutputStream dos; 
45           String lineEnd = "\r\n"; 
46           String twoHyphens = "--"; 
47           String boundary = "*****"; 
48           int bytesRead, bytesAvailable, bufferSize; 
49           byte[] buffer; 
50    
51           int maxBufferSize = 1024 * 1024; 
52    
53           File file = new File(Environment.getExternalStoragePublicDirectory(Environment.DIRECTORY_DOWNLOADS),"test.xlsx"); 
54           try { 
55               // open a URL connection to the Servlet 
56               FileInputStream fileInputStream = new FileInputStream(file); 
57               URL url = new URL(upLoadServerUri); 
58               // Open a HTTP  connection to  the URL 
59               conn = (HttpURLConnection) url.openConnection(); 
60               conn.setDoInput(true); // Allow Inputs 
61               conn.setDoOutput(true); // Allow Outputs 
62               conn.setUseCaches(false); // Don't use a Cached Copy 
63               conn.setRequestMethod("POST"); 
64               conn.setRequestProperty("Connection", "Keep-Alive"); 
65               conn.setRequestProperty("ENCTYPE", "multipart/form-data"); 
66               conn.setRequestProperty("Content-Type", "multipart/form-data;boundary=" + boundary); 
67               conn.setRequestProperty("fileToUpload", file.getName()); 
68               dos = new DataOutputStream(conn.getOutputStream()); 
69               dos.writeBytes(twoHyphens + boundary + lineEnd); 
70               dos.writeBytes("Content-Disposition: form-data; name=\"fileToUpload\";filename=\"" + file.getName() + "\"" + lineEnd); 
71               dos.writeBytes(lineEnd); 
72               // create a buffer of  maximum size 
73               bytesAvailable = fileInputStream.available(); 
74               bufferSize = Math.min(bytesAvailable, maxBufferSize); 
75               buffer = new byte[bufferSize]; 
76               // read file and write it into form... 
77               bytesRead = fileInputStream.read(buffer, 0, bufferSize); 
78    
79               while (bytesRead > 0) { 
80                   dos.write(buffer, 0, bufferSize); 
81                   bytesAvailable = fileInputStream.available(); 
82                   bufferSize = Math.min(bytesAvailable, maxBufferSize); 
83                   bytesRead = fileInputStream.read(buffer, 0, bufferSize); 
84               } 
85    
86               // send multipart form data necessary after file data... 
87               dos.writeBytes(lineEnd); 
88               dos.writeBytes(twoHyphens + boundary + twoHyphens + lineEnd); 
89    
90               // Responses from the server (code and message) 
91               serverResponseCode = conn.getResponseCode(); 
92               String serverResponseMessage = conn.getResponseMessage(); 
93               Log.i("uploadFile", "HTTP Response is : " + serverResponseMessage + ": " + serverResponseCode); 
94               if(serverResponseCode == 200){ 
95                   activity.runOnUiThread(new Runnable() { 
96                       public void run() { 
97                           Toast.makeText(activity, "File Upload Complete.", LENGTH_SHORT).show(); 
98                       } 
99                   }); 
100              } 
101   
102              //close the streams // 
103              fileInputStream.close(); 
104              dos.flush(); 
105              dos.close(); 
106          } 
107          catch (MalformedURLException ex) { 
108              //dialog.dismiss(); 
109              ex.printStackTrace(); 
110              activity.runOnUiThread(new Runnable() { 
111                  public void run() { 
112                      Toast.makeText(activity, "MalformedURLException", LENGTH_SHORT).show(); 
113                  } 
114              }); 
115              Log.e("Upload file to server", "error: " + ex.getMessage(), ex); 
116          } 
117          catch (Exception e) { 
118              //dialog.dismiss(); 
119              e.printStackTrace(); 
120              activity.runOnUiThread(new Runnable() { 
121                  public void run() { 
122                      Toast.makeText(activity, "Got Exception : see logcat ", LENGTH_SHORT).show(); 
123                  } 
124              }); 
125          } 
126          //dialog.dismiss(); 
127          //return serverResponseCode; 
128      }
